git init   初始化仓库

git add 文件   将文件添加到暂存区域

git commit -m "说明提交的内容说明"

git status  查看状态

git reset HEAD 文件名   回复上一次的操作 加~代表第几个快照

git checkout -- 文件名  恢复上一次提交的内容  （覆盖现在的）

git log 查看提交历史记录

git reset  -- head 版本快照  文件名/路径

git reset --head id号

比较暂存区与工作目录 diff --git 文件1 文件2

git rm  -f文件名  强制删除 

git mv 旧文件名  新文件名
